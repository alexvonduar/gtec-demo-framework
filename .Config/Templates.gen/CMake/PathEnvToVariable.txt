if(NOT DEFINED ENV{##ENVIRONMENT_VARIABLE_NAME##})
  message(FATAL_ERROR "The ##ENVIRONMENT_VARIABLE_NAME## environment variable is not defined")
elseif(NOT DEFINED ${##ENVIRONMENT_VARIABLE_NAME##})
  set(##ENVIRONMENT_VARIABLE_NAME## "$ENV{##ENVIRONMENT_VARIABLE_NAME##}")
  string(REPLACE "\\" "/" ##ENVIRONMENT_VARIABLE_NAME## ${##ENVIRONMENT_VARIABLE_NAME##})
endif()
